\subsubsection{Description}
\label{sec:orgc7dacdc}
\begin{itemize}
\item find all strongly connected components in O(V+E)
\item DFS two times
\begin{itemize}
\item DFS of graph produces single tree if all vertices reachable from starting point
\item else: forest
\end{itemize}
\item may find tree or forest depending on where start
\begin{itemize}
\item 0, 1, 2 -> tree
\item 3, 4 -> forest
\end{itemize}
\item \textbf{finish time of a vertex that connects to other SCCs will always be greater than finish times of vertices in the other SCC}
\begin{center}
\includegraphics[width=.9\linewidth]{./images/screenshot-03.png}
\end{center}
\end{itemize}

\subsubsection{Method}
\label{sec:org22ecb52}
\begin{enumerate}
\item DFS traversal of graph
\label{sec:orgc3394e9}
\begin{itemize}
\item create empty stack S
\item recursive DFS traversal for adjacent vertices of vertex -> push vertex to stack
\item starting from 0: 1, 2, 4, 3, 0
\end{itemize}
\item Reverse directions of all arcs -> transpose graph
\label{sec:orge9fe046}
\item DFS on transpose graph
\label{sec:orgc4467d2}
\begin{itemize}
\item while S is not empty
\begin{itemize}
\item pop vertex v from S
\item take v as source -> DFSUtil()
\item DFS from v -> strongly connected components of v
\end{itemize}
\end{itemize}
\end{enumerate}
